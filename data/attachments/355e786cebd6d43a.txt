pm.test("Status code is 200", function () {
    pm.response.to.have.status(200);
});

pm.test("Response contains pets with status 'available'", function () {
    var jsonData = pm.response.json(); // Parseamos el JSON de la respuesta
    // Verificamos que todos los objetos de la respuesta tengan el estado 'available'
    jsonData.forEach(pet => {
        pm.expect(pet.status).to.eql('available');
    });
});
